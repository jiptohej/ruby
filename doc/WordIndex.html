def index(anObject, *phrases)
  phrases.each do |aPhrase|
    aPhrase.scan /\w[-\w']+/ do |aWord| 
      aWord.downcase!
      @index[aWord] = [] if @index[aWord].nil?
      @index[aWord].push(anObject)
    end
  end
end
